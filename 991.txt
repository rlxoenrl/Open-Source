#include<stdio.h>

int main(void)
{
 int high,i,j;
 
 printf("還 偎熱 : ");
 scanf("%d",&high);
 for(i=1;i<=high;i++)
  {
   for(j=0; j<high-i; j++)
    printf(" ");
   for(j=0; j<i*2-1; j++)
    printf("*");
   printf("\n");
  }


 return 0;
}
